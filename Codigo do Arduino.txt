const int buttonPins[] = {2, 3, 4, 5, 6, 7, 8, 9};
const int numButtons = sizeof(buttonPins)  sizeof(buttonPins[0]);

int lastButtonStates[numButtons];   Store previous states

unsigned long last_time = 0;

void setup() {
  Serial.begin(9600);

  for (int i = 0; i  numButtons; i++) {
    pinMode(buttonPins[i], INPUT_PULLUP);
    lastButtonStates[i] = digitalRead(buttonPins[i]);
  }
}

void loop() {
   Heartbeat every 2 seconds (optional)
  if (millis()  last_time + 2000) {
    Serial.println(Arduino is alive!!);
    last_time = millis();
  }

  for (int i = 0; i  numButtons; i++) {
    int currentState = digitalRead(buttonPins[i]);

     Pressed HIGH - LOW
    if (lastButtonStates[i] == HIGH && currentState == LOW) {
      Serial.print(P);
      Serial.println(i);   e.g., P0 when pressed
    }

     Released LOW - HIGH
    if (lastButtonStates[i] == LOW && currentState == HIGH) {
      Serial.print(R);
      Serial.println(i);   e.g., R0 when released
    }

    lastButtonStates[i] = currentState;
  }

  delay(20);  small delay
}
